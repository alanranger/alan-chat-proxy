const questions = lines.slice(1).map(line => { const values = line.split(',').map(v => v.trim().replace(/"/g, '')); const obj = {}; headers.forEach((header, index) => { obj[header] = values[index] || ''; const result = { question: questionText, confidence, intent: data.structured?.intent || 'unknown', hasMarkdown, hasDirectAnswer, hasPills, hasContentChunks, answerLength: data.answer_markdown?.length || 0, eventsCount: data.structured?.events?.length || 0, articlesCount: data.structured?.articles?.length || 0, productsCount: data.structured?.products?.length || 0, version: data.debug?.version || 'unknown' }; 'ðŸŸ¢ High' : confidence >= 50 ? 'ðŸŸ¡ Medium' : 'ðŸ”´ Low'})`); console.log(` ðŸ“ Intent: ${result.intent}`); console.log(` ðŸ“Š Features: Markdown: ${hasMarkdown ?
